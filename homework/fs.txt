#################################################################################
# Файловая структура проекта организована по рекомендации Nested (упрощенная).	#
# Упрощения:																	#
# 	Для каждого блока своя директория, елементы не выносятся в отдельные файлы; #
# 	Реализации блока собраны в файлах с соответствующими расширениями;			#
# 	Модификаторы блоков и элементов вынесеы в отдельные файлы;					#
# 	Ненужного в директориях нет: при сборке не нужно анализировать HTML,		#
# 	все, что есть в src, используется в app;									#
# В корне src index.pug и main.sass содержат все необходимые блоки и элементы	#
#################################################################################

# Файловая структура проекта: 

project-name/
	src/
		block1/									#имена блока и директории совпадают
			block--01.png						#изображения блока, без иерархии, по порядковым номерам
			block__element--modifictor.sass		#модификатор элемента, sass
			block--modificator.sass				#модификатор блока, sass
			block.sass							#обязательные элементы блока, sass
			block.js							#обязательные элементы блока, js
		block2/	
		blockN/

	tmp/										#промежуточная папка для сборки в gulp
	app/										#готовая минимизированная сборка проекта, готовая к публикации
		fonts/
		img/
			image.png
		index.html								#бандл html
		style.min.css							#бандл css
		script.min.js							#бандл js
